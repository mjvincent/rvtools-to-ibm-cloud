# Carbon UI Network Planning Schema Specification

## Overview

This document provides detailed specifications for the network planning schema that bridges Carbon UI's planning model with Terraform generation.

```typescript
/**
 * Validate CIDR format.
 */
export function isValidCidr(cidr: string): boolean {
  const cidrRegex = /^(\d{1,3}\.){3}\d{1,3}\/\d{1,2}$/;
  if (!cidrRegex.test(cidr)) return false;

  const [ip, prefix] = cidr.split('/');
  const octets = ip.split('.').map(Number);
  const prefixNum = Number(prefix);

  return (
    octets.every(o => o >= 0 && o <= 255) &&
    prefixNum >= 0 && prefixNum <= 32
  );
}

/**
 * Validate security group rule.
 */
export function validateSecurityRule(rule: SecurityRule): string[] {
  const errors: string[] = [];

  if (rule.protocol === 'tcp' || rule.protocol === 'udp') {
    if (!rule.portMin || !rule.portMax) {
      errors.push('TCP/UDP rules require port range');
    }
    if (rule.portMin && (rule.portMin < 1 || rule.portMin > 65535)) {
      errors.push('Port min must be 1-65535');
    }
    if (rule.portMax && (rule.portMax < 1 || rule.portMax > 65535)) {
      errors.push('Port max must be 1-65535');
    }
    if (rule.portMin && rule.portMax && rule.portMin > rule.portMax) {
      errors.push('Port min must be <= port max');
    }
  }

  if (rule.direction === 'inbound' && !rule.source) {
    errors.push('Inbound rules require source');
  }

  if (rule.direction === 'outbound' && !rule.destination) {
    errors.push('Outbound rules require destination');
  }

  return errors;
}

/**
 * Validate VM assignment completeness.
 */
export function validateVmAssignment(
  assignment: VmNetworkAssignment,
  subnets: SubnetBucket[],
  securityGroups: SecurityBucket[]
): string[] {
  const errors: string[] = [];

  if (!assignment.primarySubnetId) {
    errors.push('Primary subnet required');
  } else if (!subnets.find(s => s.id === assignment.primarySubnetId)) {
    errors.push('Primary subnet not found');
  }

  if (!assignment.primarySecurityGroupId) {
    errors.push('Primary security group required');
  } else if (!securityGroups.find(sg => sg.id === assignment.primarySecurityGroupId)) {
    errors.push('Primary security group not found');
  }

  // Validate secondary NICs
  assignment.secondaryNics.forEach((nic, idx) => {
    if (!subnets.find(s => s.id === nic.subnetId)) {
      errors.push(`Secondary NIC ${idx + 1}: subnet not found`);
    }
    if (!securityGroups.find(sg => sg.id === nic.securityGroupId)) {
      errors.push(`Secondary NIC ${idx + 1}: security group not found`);
    }
  });

  return errors;
}
```

```python
def migrate_0_9_to_1_0(data: Dict[str, Any]) -> Dict[str, Any]:
    """Example migration from 0.9 to 1.0."""
    # Add new fields with defaults
```

### 5.3 Loading Network Plan (Python)

```python
# prototype/api/app.py

from models.network_planning import (
    NetworkPlanningState,
    network_planning_state_from_dict,
)

@app.get("/api/projects/{project_id}/network-plan")
async def get_network_plan(project_id: str):
    """Retrieve saved network planning state."""
    project = persistence.get_project(project_id)
    if not project:
        raise HTTPException(status_code=404, detail="Project not found")

    planning_state_json = project.get("planning_state_json", {})
    network_plan_data = planning_state_json.get("carbon_network_plan")

    if not network_plan_data:
        # Return empty network plan
        return NetworkPlanningState().__dict__

    # Convert dict to NetworkPlanningState for validation
    network_plan = network_planning_state_from_dict(network_plan_data)
    return network_plan.__dict__
```

## 7. Migration from RVTools Discovery

When a user uploads an RVTools workbook, the system should:

1. Parse RVTools `vNetwork` tab to discover source networks
2. Create initial VPC/subnet/security group buckets based on discovery
3. Allow user to refine the plan in Carbon UI
4. Use refined plan for Terraform generation

```python
# prototype/api/app.py

def create_initial_network_plan_from_rvtools(
    parsed_rvtools: Any,
    target_region: str,
    target_zone: str,
) -> NetworkPlanningState:
    """Create initial network plan from RVTools discovery."""
    unique_nets = parsed_rvtools.unique_nets

    # Create single VPC
    vpc_id = str(uuid.uuid4())
    vpc = VpcPlan(
        id=vpc_id,
        name="migration-vpc",
        region=target_region,
        address_prefix_mode="manual",
        address_prefixes=[
            AddressPrefix(
                id=str(uuid.uuid4()),
                name=f"zone-1-prefix",
                cidr="10.240.0.0/16",  # Default, user can refine
                zone=target_zone,
                is_default=True,
            )
        ],
    )

    # Create subnet for each discovered network
    subnets = []
    security_groups = []
    for idx, net in enumerate(unique_nets):
        subnet_id = str(uuid.uuid4())
        sg_id = str(uuid.uuid4())

        subnets.append(SubnetPlan(
            id=subnet_id,
            name=f"subnet-{net['name']}",
            vpc_id=vpc_id,
            zone=target_zone,
            cidr=net.get('cidr', f"10.240.{idx * 10}.0/24"),
            purpose=net.get('purpose', 'general'),
            source_network=net['name'],
        ))

        security_groups.append(SecurityGroupPlan(
            id=sg_id,
            name=f"sg-{net['name']}",
            vpc_id=vpc_id,
            purpose=net.get('purpose', 'general'),
            rules=[
                SecurityRule(
                    id=str(uuid.uuid4()),
                    direction="outbound",
                    protocol="all",
                    description="Allow all outbound",
                )
            ],
        ))

    return NetworkPlanningState(
        vpcs=[vpc],
        subnets=subnets,
        security_groups=security_groups,
        metadata=PlanningMetadata(
            target_region=target_region,
            target_zone=target_zone,
        ),
    )
```
